from fastapi import FastAPI, UploadFile from fastapi.middleware.cors import CORSMiddleware from pydantic import BaseModel from transformers import AutoTokenizer, AutoModelForCausalLM, AdamW import torch app = FastAPI() # Разрешаем фронтенду подключаться app.add_middleware( CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"], ) # Модели models = { "gpt2": AutoModelForCausalLM.from_pretrained("gpt2"), "distilgpt2": AutoModelForCausalLM.from_pretrained("distilgpt2") } tokenizers = { "gpt2": AutoTokenizer.from_pretrained("gpt2"), "distilgpt2": AutoTokenizer.from_pretrained("distilgpt2") } current_model = "gpt2" training_enabled = False device = torch.device("cuda" if torch.cuda.is_available() else "cpu") models[current_model].to(device) class Message(BaseModel): text: str @app.post("/chat") async def chat(message: Message): tokenizer = tokenizers[current_model] model = models[current_model] model.eval() inputs = tokenizer(message.text, return_tensors="pt").to(device) outputs = model.generate(**inputs, max_new_tokens=100) reply = tokenizer.decode(outputs[0], skip_special_tokens=True) return {"reply": reply} @app.post("/train") async def train(file: UploadFile): global training_enabled if not training_enabled: return {"status": "Обучение выключено"} data = await file.read() text = data.decode("utf-8") tokenizer = tokenizers[current_model] model = models[current_model] model.train() inputs = tokenizer(text, return_tensors="pt").to(device) labels = inputs.input_ids.clone() optimizer = AdamW(model.parameters(), lr=1e-5) # Один шаг обучения для демонстрации optimizer.zero_grad() outputs = model(**inputs, labels=labels) loss = outputs.loss loss.backward() optimizer.step() return {"status": f"Модель обучена на новых данных, loss={loss.item():.4f}"} @app.post("/toggle_training") async def toggle_training(): global training_enabled training_enabled = not training_enabled return {"training_enabled": training_enabled} @app.post("/switch_model") async def switch_model(model_name: str): global current_model if model_name not in models: return {"status": "Модель не найдена"} current_model = model_name models[current_model].to(device) return {"status": f"Используется модель {current_model}"}
